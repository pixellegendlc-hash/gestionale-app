# 📱 GUIDA INSTALLAZIONE GESTIONALE PWA
# Segui i passi IN ORDINE

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
PASSO 1 — CREA ACCOUNT GITHUB (gratis)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Vai su https://github.com
2. Clicca "Sign up"
3. Inserisci email, password, username
4. Conferma l'email

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
PASSO 2 — CREA REPOSITORY PER I DATI
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Questo repo conterrà i tuoi dati (task, finanze, appunti).

1. Clicca il "+" in alto a destra → "New repository"
2. Nome: gestionale-dati
3. Seleziona "Private" (privato!)
4. Spunta "Add a README file"
5. Clicca "Create repository"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
PASSO 3 — CREA REPOSITORY PER IL CODICE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Questo repo conterrà il codice dell'app.

1. Clicca il "+" → "New repository"
2. Nome: gestionale-app
3. Seleziona "Public"
4. Spunta "Add a README file"
5. Clicca "Create repository"

Ora carica i file dell'app:
6. Clicca "Add file" → "Upload files"
7. Trascina TUTTI i file della cartella gestionale_pwa
   (app.py, requirements.txt, Procfile, la cartella templates/, la cartella static/)
8. Clicca "Commit changes"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
PASSO 4 — CREA TOKEN GITHUB
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Il token permette all'app di leggere/scrivere i dati.

1. Clicca sulla tua foto in alto a destra → "Settings"
2. Scorri in fondo → "Developer settings" (ultima voce)
3. "Personal access tokens" → "Tokens (classic)"
4. "Generate new token" → "Generate new token (classic)"
5. Nome: gestionale-token
6. Scadenza: "No expiration"
7. Spunta "repo" (la prima voce, seleziona tutto il gruppo)
8. Clicca "Generate token"
9. ⚠️ COPIA IL TOKEN ORA — non lo vedrai più!
   Salvalo in un posto sicuro (es. blocco note)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
PASSO 5 — DEPLOY SU RENDER (gratis)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Vai su https://render.com
2. Clicca "Get started for free"
3. Registrati con GitHub (clicca "Continue with GitHub")
4. Autorizza Render

5. Clicca "New +" → "Web Service"
6. Seleziona il repository "gestionale-app"
7. Impostazioni:
   - Name: gestionale
   - Runtime: Python 3
   - Build Command: pip install -r requirements.txt
   - Start Command: gunicorn app:app
   - Instance Type: Free

8. Scorri in basso → "Environment Variables"
   Aggiungi queste 3 variabili (clicca "Add Environment Variable"):

   Key: GITHUB_TOKEN
   Value: [incolla il token copiato al passo 4]

   Key: GITHUB_OWNER
   Value: [il tuo username GitHub]

   Key: GITHUB_REPO
   Value: gestionale-dati

9. Clicca "Create Web Service"
10. Aspetta 2-3 minuti che finisca il deploy

11. In alto vedrai un link tipo: https://gestionale.onrender.com
    Questo è il tuo gestionale online! 🎉

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
PASSO 6 — AGGIUNGI ALLA SCHERMATA HOME
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Su ANDROID (Chrome):
1. Apri il link dal browser Chrome
2. Tocca i 3 puntini in alto a destra
3. "Aggiungi a schermata Home"
4. Conferma → appare come app!

Su IPHONE (Safari):
1. Apri il link da Safari
2. Tocca il tasto "Condividi" (quadrato con freccia)
3. "Aggiungi a schermata Home"
4. Conferma → appare come app!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
